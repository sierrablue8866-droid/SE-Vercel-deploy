        // ============================================================
        // 1. COMPOUND DICTIONARY

        const compoundCodes = {
            'ميفيدا': 'MI',
            'mivida': 'MI',
            'ماونتن فيو': 'MV',
            'mountain view': 'MV',
            'هايد بارك': 'HP',
            'hyde park': 'HP',

            'الرحاب': 'RH',
            'rehab': 'RH',
            'الشروق': 'ES',
            'el shorouk': 'ES',
            'المقصد': 'MQ',
            'almaksad': 'MQ',
            'دار مصر': 'DM',
            'dar misr': 'DM'
        };

        // ============================================================
        // 2. EXTRACTION FUNCTIONS
        // ============================================================
        function extractCompound(text) {
            text = text.toLowerCase();
            for (let [name, code] of Object.entries(compoundCodes)) {
                if (text.includes(name.toLowerCase())) {
                    return { name: name, code: code };
                }
            }
            let words = text.split(' ').filter(w => w.length > 0);
            if (words.length > 0) {
                let fallback = words[0].substring(0, 2).toUpperCase();
                return { name: words[0], code: fallback };
            }
            return null;
        }

        function extractNumber(text, keywords) {
            let patterns = [
                new RegExp(`(\\d+)\\s*(${keywords.join('|')})`, 'i'),
                new RegExp(`(${keywords.join('|')})\\s*(\\d+)`, 'i'),
                /(\d+)/i
            ];
            for (let pattern of patterns) {
                let match = text.match(pattern);
                if (match) return parseInt(match[1] || match[2]);
            }
            return null;
        }

        function extractBedrooms(text) {
            if (text.includes('استوديو') || text.includes('studio')) return 1;
            if (text.includes('غرفتين')) return 2;
            if (text.includes('ثلاث')) return 3;
            if (text.includes('أربع')) return 4;
            let words = ['غرف', 'bedrooms', 'bedroom', 'br', 'غرفة', 'room'];
            let num = extractNumber(text, words);
            if (num) return num;
            return null;
        }

        function extractBathrooms(text) {
            let words = ['حمام', 'bathrooms', 'bathroom', 'bath'];
            return extractNumber(text, words);
        }

        function extractArea(text) {
            let match = text.match(/(\d+)\s*(متر|sqm|m²|m)/i);
            return match ? parseInt(match[1]) : null;
        }

        function extractFurnished(text) {
            if (text.includes('مفروش') || text.includes('furnished')) return 'F';
            if (text.includes('نصف') || text.includes('semi')) return 'S';
            if (text.includes('مطبخ') || text.includes('kitchen')) return 'K';
            if (text.includes('غير مفروش') || text.includes('unfurnished')) return 'U';
            return null;
        }

        function extractPrice(text) {
            let arabicToEnglish = { '٠': '0', '١': '1', '٢': '2', '٣': '3', '٤': '4', '٥': '5', '٦': '6', '٧': '7', '٨': '8',
                '٩': '9' };
            let cleaned = text;
            for (let [a, e] of Object.entries(arabicToEnglish)) {
                cleaned = cleaned.replace(new RegExp(a, 'g'), e);
            }
            let match = cleaned.match(/(\d+[.,]?\d*)\s*(مليون|م|million)?\s*(الف|k|ألف)?/i);
            if (!match) {
                let simpleMatch = cleaned.match(/(\d{4,})/);
                if (simpleMatch) {
                    let val = parseInt(simpleMatch[1]);
                    return { value: val, currency: 'EGP' };
                }
                return null;
            }
            let value = parseFloat(match[1].replace(',', ''));

// 1. Extract Compound
function extractCompound(text) {
    const compoundCodes = {
        'ميفيدا': 'MI', 'mivida': 'MI',
        'ماونتن فيو': 'MV', 'mountain view': 'MV',
        'فيليت': 'VS', 'villette': 'VS',
        'كايرو فيستيفال': 'CFC', 'cairo festival': 'CFC',
        'هايد بارك': 'HP', 'hyde park': 'HP',
        'ليك فيو': 'LV', 'lake view': 'LV'
    };
    text = text.toLowerCase();
    for (let [name, code] of Object.entries(compoundCodes)) {
        if (text.includes(name.toLowerCase())) {
            return { name: name, code: code };
        }
    }
    // Fallback: first two letters
    let words = text.split(' ').filter(w => w.length > 0);
    if (words.length > 0) {
        return { name: words[0], code: words[0].substring(0, 2).toUpperCase() };
    }
    return null;
}

// 2. Extract Bedrooms
function extractBedrooms(text) {
    if (text.includes('استوديو') || text.includes('studio')) return 1;
    if (text.includes('غرفتين')) return 2;
    if (text.includes('ثلاث')) return 3;
    if (text.includes('أربع')) return 4;
    let match = text.match(/(\d+)\s*(غرف|bedrooms?|br)/i);
    return match ? parseInt(match[1]) : null;
}

// 3. Extract Bathrooms
function extractBathrooms(text) {
    let match = text.match(/(\d+)\s*(حمام|bathrooms?|bath)/i);
    return match ? parseInt(match[1]) : null;
}

// 4. Extract Area
function extractArea(text) {
    let match = text.match(/(\d+)\s*(متر|sqm|m²|m)/i);
    return match ? parseInt(match[1]) : null;
}

// 5. Extract Furnished Status
function extractFurnished(text) {
    if (text.includes('مفروش') || text.includes('furnished')) return 'F';
    if (text.includes('نصف') || text.includes('semi')) return 'S';
    if (text.includes('مطبخ') || text.includes('kitchen')) return 'K';
    if (text.includes('غير مفروش') || text.includes('unfurnished')) return 'U';
    return null;
}

// 6. Extract Price
function extractPrice(text) {
    // Convert Arabic numbers
    const arabicToEnglish = { '٠':'0', '١':'1', '٢':'2', '٣':'3', '٤':'4',
                             '٥':'5', '٦':'6', '٧':'7', '٨':'8', '٩':'9' };
    let cleaned = text;
    for (let [a, e] of Object.entries(arabicToEnglish)) {
        cleaned = cleaned.replace(new RegExp(a, 'g'), e);
    }
    let match = cleaned.match(/(\d+[.,]?\d*)\s*(مليون|م|million)?\s*(الف|k|ألف)?/i);
    if (!match) return null;
    let value = parseFloat(match[1].replace(',', ''));
    if (match[2] && match[2].includes('مليون')) value *= 1000000;
    else if (match[3] && match[3].includes('الف')) value *= 1000;
    let currency = 'EGP';
    if (text.includes('$') || text.includes('dollar')) currency = 'USD';
    if (text.includes('درهم') || text.includes('aed')) currency = 'AED';
    return { value, currency };
}
